📋 คู่มือการติดตั้ง Google Apps Script

ขั้นตอนการตั้งค่า:

  1. สร้าง Google Sheet ใหม่ แล้วสร้าง Sheets ดังนี้: Products, StockIn, StockOut, Inventory
  2. เปิด Extensions → Apps Script
  3. วางโค้ดด้านล่างลงใน Code.gs แล้วกด Save
  4. คลิก Deploy → New Deployment → Web App
  5. ตั้ง Execute as: Me, Who has access: Anyone
  6. คัดลอก Web App URL มาวางในช่อง URL ด้านบน

const SHEET_ID = '1PbJirlvKx1tqrhPJt2guZkFdkUHMRTWmiVekGgq2seU';

function doGet(e) {
  return handleRequest(e);
}
function doPost(e) {
  return handleRequest(e);
}

function handleRequest(e) {
  const action = e.parameter.action || (e.postData ? JSON.parse(e.postData.contents).action : '');
  const data   = e.postData ? JSON.parse(e.postData.contents) : e.parameter;
  let result;
  try {
    switch(action) {
      case 'getProducts':   result = getProducts(); break;
      case 'addProduct':    result = addProduct(data); break;
      case 'stockIn':       result = stockIn(data); break;
      case 'stockOut':      result = stockOut(data); break;
      case 'getInventory':  result = getInventory(); break;
      case 'getReport':     result = getReport(); break;
      case 'ping':          result = {ok:true}; break;
      default: result = {error:'Unknown action'};
    }
  } catch(err) {
    result = {error: err.message};
  }
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function getProducts() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sh = ss.getSheetByName('Products') || ss.insertSheet('Products');
  const data = sh.getDataRange().getValues();
  if(data.length <= 1) return [];
  return data.slice(1).map(r => ({id:r[0],name:r[1],unit:r[2],minQty:r[3],category:r[4]}));
}

function addProduct(d) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sh = ss.getSheetByName('Products') || ss.insertSheet('Products');
  if(sh.getLastRow() === 0) sh.appendRow(['ID','ชื่อสินค้า','หน่วย','จำนวนขั้นต่ำ','หมวดหมู่']);
  const id = 'P' + String(Date.now()).slice(-6);
  sh.appendRow([id,d.name,d.unit,d.minQty||0,d.category||'']);
  return {ok:true, id};
}

function stockIn(d) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sh = ss.getSheetByName('StockIn') || ss.insertSheet('StockIn');
  if(sh.getLastRow() === 0) sh.appendRow(['วันที่','รหัสสินค้า','ชื่อสินค้า','จำนวน','ผู้บันทึก','หมายเหตุ']);
  sh.appendRow([new Date(),d.productId,d.productName,Number(d.qty),d.recorder||'',d.note||'']);
  updateInventory(d.productId, Number(d.qty));
  return {ok:true};
}

function stockOut(d) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const inv = getInventoryMap();
  const current = inv[d.productId] || 0;
  if(current < Number(d.qty)) return {error:'สินค้าไม่เพียงพอ (คงเหลือ ' + current + ')'};
  const sh = ss.getSheetByName('StockOut') || ss.insertSheet('StockOut');
  if(sh.getLastRow() === 0) sh.appendRow(['วันที่','รหัสสินค้า','ชื่อสินค้า','จำนวน','ผู้เบิก','แผนก','หมายเหตุ']);
  sh.appendRow([new Date(),d.productId,d.productName,Number(d.qty),d.requester||'',d.dept||'',d.note||'']);
  updateInventory(d.productId, -Number(d.qty));
  return {ok:true};
}

function getInventoryMap() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sh = ss.getSheetByName('Inventory');
  if(!sh) return {};
  const data = sh.getDataRange().getValues();
  const map = {};
  data.slice(1).forEach(r => { map[r[0]] = r[2]; });
  return map;
}

function getInventory() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const invSh = ss.getSheetByName('Inventory');
  const prods = getProducts();
  const prodMap = {};
  prods.forEach(p => { prodMap[p.id] = p; });
  if(!invSh) return [];
  const data = invSh.getDataRange().getValues();
  return data.slice(1).map(r => ({
    productId:r[0], productName:r[1], qty:r[2],
    updatedAt: r[3] ? new Date(r[3]).toLocaleDateString('th-TH') : '',
    minQty: prodMap[r[0]]?.minQty || 0,
    unit: prodMap[r[0]]?.unit || '',
    category: prodMap[r[0]]?.category || ''
  }));
}

function getReport() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const inSh = ss.getSheetByName('StockIn');
  const outSh = ss.getSheetByName('StockOut');
  const inData = inSh ? inSh.getDataRange().getValues().slice(1) : [];
  const outData = outSh ? outSh.getDataRange().getValues().slice(1) : [];
  const now = new Date();
  const monthStart = new Date(now.getFullYear(), now.getMonth(), 1);
  const monthIn = inData.filter(r => new Date(r[0]) >= monthStart).reduce((s,r) => s+r[3], 0);
  const monthOut = outData.filter(r => new Date(r[0]) >= monthStart).reduce((s,r) => s+r[3], 0);
  return { totalIn: inData.reduce((s,r)=>s+r[3],0), totalOut: outData.reduce((s,r)=>s+r[3],0), monthIn, monthOut,
    recentIn: inData.slice(-5).reverse(), recentOut: outData.slice(-5).reverse() };
}
